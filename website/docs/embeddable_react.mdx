import { customFields } from '../docusaurus.config.json'

export function Link() {
  const { currentLink } = customFields
  const pathArray = currentLink.split('/')
  const version = pathArray[pathArray.length - 2]

  return (
    <a href={`https://jbrowse.org/storybook/${version}/`}>Storybook Docs</a>
  )
}
